**build.py** — automated PyInstaller build script:

```python
#!/usr/bin/env python3
"""Build the Lumora Python AI sidecar into a standalone Windows executable using PyInstaller.

Usage:
    py build.py              # Build the sidecar binary
    py build.py --clean      # Clean build artifacts then build
    py build.py --check      # Only verify dependencies, don't build

Output: ../binaries/lumora-sidecar-x86_64-pc-windows-msvc.exe
"""

import os, sys, shutil, subprocess, argparse
from pathlib import Path

# Project paths
SIDECAR_DIR = Path(__file__).resolve().parent
PROJECT_DIR = SIDECAR_DIR.parent  # src-tauri/
BINARIES_DIR = PROJECT_DIR / "binaries"
BUILD_OUTPUT_DIR = SIDECAR_DIR / "dist" / "lumora-sidecar"
BUILD_WORK_DIR = SIDECAR_DIR / "build"

# Target triple for Windows MSVC (must match Tauri's build target)
TARGET_TRIPLE = "x86_64-pc-windows-msvc"
OUTPUT_NAME = f"lumora-sidecar-{TARGET_TRIPLE}.exe"

def check_python_version():
    """Verify Python version is compatible with PyTorch 2.12.1.
    
    PyTorch 2.12.1 supports Python 3.10-3.13. Python 3.14 wheels may not be available yet.
    See RESEARCH.md Open Question 3.
    """
    version = sys.version_info
    if version.major == 3 and version.minor <= 13:
        print(f"[build] Python {version.major}.{version.minor}.{version.micro} is compatible.")
        return True
    elif version.major == 3 and version.minor >= 14:
        print(f"[build] WARNING: Python {version.major}.{version.minor} detected.")
        print("[build] PyTorch 2.12.1 may not have wheels for Python 3.14+.")
        print("[build] If pip install fails, use Python 3.13 instead.")
        # Continue anyway — let pip decide if wheels exist
        return True
    else:
        print(f"[build] ERROR: Python {version.major}.{version.minor} is not supported.")
        print("[build] Required: Python 3.10-3.13")
        return False

def clean_build_artifacts():
    """Remove previous PyInstaller output directories."""
    for dirpath in [BUILD_OUTPUT_DIR.parent, BUILD_WORK_DIR]:
        if dirpath.exists():
            print(f"[build] Cleaning {dirpath}")
            shutil.rmtree(dirpath)
    # Also clean PyInstaller .spec file if it exists
    spec_file = SIDECAR_DIR / "lumora-sidecar.spec"
    if spec_file.exists():
        spec_file.unlink()

def verify_dependencies():
    """Check that all required packages are importable."""
    packages = {
        "open_clip": "open-clip-torch",
        "torch": "torch (CPU)",
        "PIL": "Pillow",
        "PyInstaller": "pyinstaller",
    }
    missing = []
    for module, package in packages.items():
        try:
            __import__(module)
            print(f"[build]   OK: {package}")
        except ImportError:
            print(f"[build]   MISSING: {package}")
            missing.append(package)
    
    if missing:
        print(f"\n[build] Missing packages: {', '.join(missing)}")
        print("[build] Install with:")
        print("[build]   pip install torch --index-url https://download.pytorch.org/whl/cpu")
        print(f"[build]   pip install {' '.join(m.replace('torch (CPU)', '').strip() for m in missing if 'torch' not in m)}")
        return False
    return True

def run_pyinstaller():
    """Run PyInstaller with the correct flags for the sidecar.
    
    Flags rationale (traceable to RESEARCH.md):
    - --onedir: avoids 30s cold start and 2GB binary (Pitfall 1)
    - --console: keeps stdin/stdout working on Windows (Pitfall 2)
    - --name: base name without target triple (Tauri appends it)
    - --hidden-import: ensures open_clip submodules are bundled
    - --exclude-module: strips unused torch subsystems to reduce size
    - --add-data: bundles requirements.txt for reference
    """
    print("[build] Running PyInstaller...")
    
    pyinstaller_args = [
        sys.executable, "-m", "PyInstaller",
        "--onedir",
        "--console",
        "--name", "lumora-sidecar",
        "--distpath", str(BUILD_OUTPUT_DIR.parent),
        "--workpath", str(BUILD_WORK_DIR),
        "--specpath", str(SIDECAR_DIR),
        "--add-data", f"requirements.txt{os.pathsep}.",
        "--hidden-import", "open_clip",
        "--hidden-import", "open_clip.factory",
        "--hidden-import", "open_clip.model",
        "--hidden-import", "open_clip.tokenizer",
        "--hidden-import", "open_clip.transform",
        "--hidden-import", "open_clip.pretrained",
        "--hidden-import", "huggingface_hub",
        "--exclude-module", "torch.distributed",
        "--exclude-module", "torch.utils.tensorboard",
        "--exclude-module", "torchvision",
        "main.py",
    ]
    
    result = subprocess.run(pyinstaller_args, cwd=str(SIDECAR_DIR))
    
    if result.returncode != 0:
        print("[build] PyInstaller failed!")
        return False
    
    # Find the built executable
    built_exe = BUILD_OUTPUT_DIR / "lumora-sidecar.exe"
    if not built_exe.exists():
        print(f"[build] ERROR: Expected output not found at {built_exe}")
        return False
    
    # Copy and rename to Tauri's expected path with target triple suffix
    os.makedirs(BINARIES_DIR, exist_ok=True)
    target_path = BINARIES_DIR / OUTPUT_NAME
    shutil.copy2(built_exe, target_path)
    
    print(f"[build] Built: {target_path}")
    
    # Report binary size
    size_mb = target_path.stat().st_size / (1024 * 1024)
    print(f"[build] Binary size: {size_mb:.1f} MB")
    if size_mb > 500:
        print("[build] WARNING: Binary exceeds 500MB — check for CUDA torch inclusion!")
    
    return True

def main():
    parser = argparse.ArgumentParser(description="Build Lumora Python AI sidecar")
    parser.add_argument("--clean", action="store_true", help="Clean previous build artifacts")
    parser.add_argument("--check", action="store_true", help="Only verify dependencies, don't build")
    args = parser.parse_args()
    
    print("[build] Lumora Sidecar Build Script")
    print(f"[build] Python: {sys.version}")
    print(f"[build] Sidecar dir: {SIDECAR_DIR}")
    print(f"[build] Output: {BINARIES_DIR / OUTPUT_NAME}")
    print()
    
    if not check_python_version():
        sys.exit(1)
    
    if args.clean:
        clean_build_artifacts()
    
    print("[build] Checking dependencies...")
    if not verify_dependencies():
        print("\n[build] Please install missing dependencies, then re-run.")
        sys.exit(1)
    
    if args.check:
        print("\n[build] All dependencies OK. Run without --check to build.")
        return
    
    print()
    if run_pyinstaller():
        print(f"\n[build] SUCCESS: {BINARIES_DIR / OUTPUT_NAME}")
        print("[build] Tauri will spawn this binary as a sidecar on app startup.")
    else:
        print("\n[build] FAILED: See errors above.")
        sys.exit(1)

if __name__ == "__main__":
    main()
```

Key design:
- `build.py` is self-documenting — prints what it's doing at each step
- Python version check warns about 3.14 but doesn't block (per RESEARCH.md Open Question 3)
- `--clean` flag for CI and rebuilds — removes stale PyInstaller cache
- `--check` flag for CI dependency verification without building
- Binary size warning at 500MB threshold (per RESEARCH.md Pitfall 1 — CPU torch should produce ~150MB binary)
- Output naming with target triple suffix (per RESEARCH.md Pitfall 3)
- `--exclude-module torch.distributed` and `torch.utils.tensorboard` reduce binary size

Do NOT:
- Use `--onefile` mode (Pitfall 1 in RESEARCH.md)
- Hardcode absolute paths — everything is relative to build.py's location
- Bundle the model checkpoint in the PyInstaller binary
- Add CUDA-related hidden imports

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| pip→PyPI | Package downloads cross network — supply chain risk for all 5 Python packages |
| PyInstaller→filesystem | PyInstaller reads all imported modules and writes EXE — could bundle unexpected code |
| Built binary→network | First launch downloads 605MB model checkpoint from Hugging Face Hub |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-008-16 | Tampering | pip install (supply chain) | mitigate | Blocking human checkpoint (Task 2) for all [ASSUMED] packages — user manually verifies each on pypi.org before install |
| T-008-17 | Tampering | PyInstaller hidden imports | mitigate | --hidden-import list is explicit; PyInstaller only bundles what's imported; clean venv prevents pulling in unrelated packages |
| T-008-18 | Elevation of Privilege | Built EXE | accept | Binary runs at user privilege level; no elevation requested; stdin/stdout communication is local-only |
| T-008-19 | Denial of Service | Model download on first launch | accept | 600s timeout in download_model.py (Plan 01); if download fails, sidecar starts but model_loaded=false; controlled by Plan 01 mitigation |
| T-008-SC | Tampering | pip install from PyPI | mitigate | slopcheck unavailable — ALL packages [ASSUMED]; Task 2 checkpoint gates every install behind manual verification |
