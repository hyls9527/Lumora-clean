```python
import uuid
import threading
from concurrent.futures import ThreadPoolExecutor, as_completed
from model import generate_embedding

class BatchProcessor:
    """Processes batches of images concurrently using a thread pool.
    
    max_workers=4 per PY-04 requirement. Inference on the CLIP model
    is serialized via a threading.Lock to prevent PyTorch thread-safety issues
    (RESEARCH.md Pitfall 5).
    """
    
    def __init__(self, max_workers=4):
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.model_lock = threading.Lock()
        self.active_batches = {}
        self._lock = threading.Lock()
    
    def submit_batch(self, image_paths, progress_callback):
        """Submit a batch of images for embedding.
        
        Args:
            image_paths: list[str] — absolute paths to image files
            progress_callback: callable(path, current, total) — called after each image completes
        
        Returns:
            batch_id: str — unique identifier for this batch
        """
        batch_id = str(uuid.uuid4())
        total = len(image_paths)
        results = {}
        errors = {}
        futures = {}
        
        def embed_with_lock(image_path):
            """Wrapper that acquires model_lock before calling generate_embedding."""
            try:
                with self.model_lock:
                    embedding = generate_embedding(image_path)
                return ("success", image_path, embedding)
            except Exception as e:
                return ("error", image_path, str(e))
        
        # Submit all images to the thread pool — futures run concurrently
        for path in image_paths:
            future = self.executor.submit(embed_with_lock, path)
            futures[future] = path
        
        with self._lock:
            self.active_batches[batch_id] = {
                "futures": futures,
                "total": total,
                "completed": 0,
                "results": results,
                "errors": errors,
                "callback": progress_callback,
            }
        
        # Start a daemon thread to watch for completions
        watcher = threading.Thread(
            target=self._watch_batch,
            args=(batch_id,),
            daemon=True
        )
        watcher.start()
        
        return batch_id
    
    def _watch_batch(self, batch_id):
        """Monitor futures for completion and invoke progress callbacks.
        Runs in a daemon thread — does not block the main JSON-RPC loop.
        """
        with self._lock:
            batch = self.active_batches.get(batch_id)
            if not batch:
                return
        
        futures = batch["futures"]
        callback = batch["callback"]
        
        try:
            for future in as_completed(futures.keys()):
                status, path, data = future.result()
                
                with self._lock:
                    if status == "success":
                        batch["results"][path] = data
                    else:
                        batch["errors"][path] = data
                    
                    batch["completed"] += 1
                    current = batch["completed"]
                    total = batch["total"]
                
                # Call progress callback OUTSIDE the lock to avoid deadlocks
                if callback:
                    try:
                        callback(path, current, total)
                    except Exception:
                        pass  # Don't let a callback failure crash the batch
        except Exception:
            pass  # Future cancellation or executor shutdown
    
    def get_results(self, batch_id):
        """Get the results for a completed batch.
        
        Returns:
            dict with keys: 'embeddings' (dict path->list[float]), 'errors' (dict path->str),
            'completed' (int), 'total' (int)
        """
        with self._lock:
            batch = self.active_batches.get(batch_id)
            if not batch:
                return None
            
            return {
                "embeddings": dict(batch["results"]),
                "errors": dict(batch["errors"]),
                "completed": batch["completed"],
                "total": batch["total"],
            }
    
    def is_complete(self, batch_id):
        """Check if a batch has finished processing all images."""
        with self._lock:
            batch = self.active_batches.get(batch_id)
            if not batch:
                return True
            return batch["completed"] >= batch["total"]
    
    def shutdown(self):
        """Gracefully shut down the thread pool. Waits for running jobs."""
        self.executor.shutdown(wait=True)
```

Key design decisions (traceable to RESEARCH.md):
- `max_workers=4` per PY-04 requirement and RESEARCH.md Pattern 3
- `threading.Lock` around `model.encode_image()` per Pitfall 5 — prevents non-deterministic crashes
- `embed_with_lock` wrapper acquires the lock THEN calls `generate_embedding` — the function itself doesn't need to know about threading
- Progress callback invoked OUTSIDE the lock — avoids deadlock if callback tries to access batch state
- Daemon watcher threads — if main process exits, watchers die without blocking shutdown
- Per-image error isolation — if one image fails, the batch continues (error recorded in `errors` dict)
- `as_completed()` processes results in completion order, not submission order — faster feedback
- `shutdown(wait=True)` — waits for in-flight jobs to finish gracefully

Do NOT:
- Use multiprocessing (each worker would need its own 605MB model copy)
- Pre-allocate result arrays (images complete in arbitrary order)
- Block the main thread in `submit_batch` — it returns immediately
- Log progress to stderr from within batch.py (the callback in main.py handles output)

```python
# Register embed_batch handler (PY-04)
# Import here to avoid circular imports and defer ThreadPoolExecutor creation
from batch import BatchProcessor
_batch_processor = BatchProcessor(max_workers=4)

def handle_embed_batch(params):
    """JSON-RPC handler for embed_batch method (PY-04).
    
    Expects params: {"image_paths": ["path1", "path2", ...]}
    Returns immediately with batch_id. Progress notifications are sent
    as JSON-RPC notifications (no id) to stdout.
    """
    image_paths = params.get("image_paths", [])
    if not image_paths:
        raise ValueError("image_paths must be a non-empty list")
    
    # Create a progress callback that writes JSON-RPC notifications to stdout
    def progress_callback(image_path, current, total):
        notification = {
            "jsonrpc": "2.0",
            "method": "batch_progress",
            "params": {
                "batch_id": batch_id,
                "current": current,
                "total": total,
                "current_image": image_path
            }
        }
        # Write directly to stdout — safe because main thread handles I/O serially
        import sys, json
        sys.stdout.write(json.dumps(notification) + "\n")
        sys.stdout.flush()
    
    batch_id = _batch_processor.submit_batch(image_paths, progress_callback)
    return {
        "batch_id": batch_id,
        "total": len(image_paths),
        "status": "processing"
    }

server.register("embed_batch", handle_embed_batch)

# Also register a method to poll batch results (for Phase 009 integration)
def handle_batch_status(params):
    """JSON-RPC handler for batch_status — check if a batch is done."""
    batch_id = params.get("batch_id")
    if not batch_id:
        raise ValueError("batch_id is required")
    results = _batch_processor.get_results(batch_id)
    is_complete = _batch_processor.is_complete(batch_id)
    return {
        "batch_id": batch_id,
        "complete": is_complete,
        "results": results
    }

server.register("batch_status", handle_batch_status)
```

Note: The `main.py` modifications should be placed AFTER the existing handler registrations but BEFORE `server.run()`. The re-import of `sys, json` inside the progress callback is intentional — it ensures the callback has access to the modules even when called from a different thread.

```python
import sys, os, time, json, pytest
from unittest.mock import patch, MagicMock

sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..'))

@pytest.fixture
def mock_embedding():

def test_concurrent_processing(mock_embedding):
    """PY-04: 4 images processed concurrently — wall time < sum of individual times."""
    with patch('batch.generate_embedding', side_effect=mock_embedding):
        from batch import BatchProcessor
        
        bp = BatchProcessor(max_workers=4)
        paths = ["img1.jpg", "img2.jpg", "img3.jpg", "img4.jpg"]
        progress_items = []
        
        def track_progress(path, current, total):
            progress_items.append((path, current, total))
        
        start = time.time()
        batch_id = bp.submit_batch(paths, track_progress)
        
        # Wait for completion (poll)
        timeout = 5
        while not bp.is_complete(batch_id) and timeout > 0:
            time.sleep(0.05)
            timeout -= 0.05
        
        elapsed = time.time() - start
        
        # Serial would be ~400ms (4 * 100ms); concurrent should be ~100-150ms
        assert elapsed < 0.3, f"Concurrent processing too slow: {elapsed:.2f}s (expected <0.3s)"
        
        results = bp.get_results(batch_id)
        assert results["completed"] == 4
        assert len(results["embeddings"]) == 4
        
        bp.shutdown()

def test_progress_notifications(mock_embedding):
    """PY-04: Progress callback invoked per-item with correct current/total."""
    with patch('batch.generate_embedding', side_effect=mock_embedding):
        from batch import BatchProcessor
        
        bp = BatchProcessor(max_workers=2)
        paths = ["a.jpg", "b.jpg", "c.jpg"]
        progress = []
        
        batch_id = bp.submit_batch(paths, lambda p, c, t: progress.append((c, t)))
        
        timeout = 5
        while not bp.is_complete(batch_id) and timeout > 0:
            time.sleep(0.05)
            timeout -= 0.05
        
        assert len(progress) == 3, f"Expected 3 progress callbacks, got {len(progress)}"
        # Last callback should show 3/3
        assert progress[-1] == (3, 3)
        
        bp.shutdown()

def test_error_isolation(mock_embedding):
    """PY-04: One failing image doesn't block the rest of the batch."""
    call_count = [0]
    def flaky_embed(image_path):
        call_count[0] += 1
        if "bad" in image_path:
            raise ValueError("Simulated processing error")
        return [0.5] * 512
    
    with patch('batch.generate_embedding', side_effect=flaky_embed):
        from batch import BatchProcessor
        
        bp = BatchProcessor(max_workers=2)
        paths = ["good1.jpg", "bad.jpg", "good2.jpg"]
        
        batch_id = bp.submit_batch(paths, None)
        
        timeout = 5
        while not bp.is_complete(batch_id) and timeout > 0:
            time.sleep(0.05)
            timeout -= 0.05
```
