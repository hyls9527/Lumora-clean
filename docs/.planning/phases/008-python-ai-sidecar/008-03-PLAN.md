**conftest.py** — shared fixtures:
```python
import sys, os, tempfile, pytest
from PIL import Image

# Ensure sidecar/ is importable
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..'))

@pytest.fixture
def temp_image():
    """Create a temporary 224x224 RGB JPEG image for testing.
    Automatically deleted after the test.
    """
    with tempfile.NamedTemporaryFile(suffix='.jpg', delete=False) as f:
        img = Image.new('RGB', (224, 224), color=(128, 64, 32))
        img.save(f, format='JPEG')
        temp_path = f.name
    yield temp_path
    os.unlink(temp_path)

@pytest.fixture
def temp_large_image():
    """Create a fake 'large' image file for testing size rejection (>50MB)."""
    with tempfile.NamedTemporaryFile(suffix='.jpg', delete=False) as f:
        f.seek(51 * 1024 * 1024 - 1)  # 51MB
        f.write(b'\x00')
        temp_path = f.name
    yield temp_path
    os.unlink(temp_path)

@pytest.fixture
def mock_model():
    """Mock CLIP model that returns a fixed 512-dim embedding.
    Use this to avoid loading the real 605MB model in unit tests.
    """
    import torch
    class MockModel:
        def eval(self):
            return self
        def encode_image(self, image_tensor):
            # Return a deterministic embedding based on input shape
            batch_size = image_tensor.shape[0]
            return torch.ones(batch_size, 512) * 0.5
    
    class MockPreprocess:
        def __call__(self, image):
            return torch.randn(3, 224, 224)
    
    return MockModel(), MockPreprocess()
```

Key design:
- `temp_image` fixture uses `tempfile.NamedTemporaryFile` with `delete=False` and manual cleanup — avoids Windows file-locking issues.
- `mock_model` returns a `MockModel` that produces deterministic 512-dim embeddings — no network, no 605MB download, no GPU. Tests run in <1 second.
- `conftest.py` handles sys.path setup so individual test files don't need to repeat it.
- Pillow is used to create the temp image — PIL is already a dependency (listed in Plan 05 requirements.txt).

Do NOT: create fixtures that depend on the real CLIP model (too slow for unit tests), use hardcoded absolute paths, import torch at module level (delays test collection when torch isn't installed yet).

**test_jsonrpc.py** — JSONRPCServer tests (6 test cases):
```python
import json, io, sys
import pytest
from jsonrpc import JSONRPCServer

@pytest.fixture
def server():
    s = JSONRPCServer()
    s.register("echo", lambda params: params)
    s.register("add", lambda params: {"sum": params["a"] + params["b"]})
    return s

@pytest.fixture
def capture_stdout():
    """Redirect stdout to StringIO for capturing JSON-RPC responses."""
    captured = io.StringIO()
    old_stdout = sys.stdout
    sys.stdout = captured
    yield captured
    sys.stdout = old_stdout

def test_valid_request_dispatches_to_handler(server, capture_stdout):
    """PY-02: Valid request returns id-matched result."""
    request = json.dumps({"jsonrpc":"2.0","method":"echo","params":{"msg":"hello"},"id":1})
    sys.stdin = io.StringIO(request)
    server.run()
    response = json.loads(capture_stdout.getvalue().strip())
    assert response["jsonrpc"] == "2.0"
    assert response["result"] == {"msg": "hello"}
    assert response["id"] == 1

def test_parse_error_on_invalid_json(server, capture_stdout):
    """PY-02: Invalid JSON returns parse error -32700."""
    sys.stdin = io.StringIO("not valid json{{{")
    server.run()
    response = json.loads(capture_stdout.getvalue().strip())
    assert response["jsonrpc"] == "2.0"
    assert response["error"]["code"] == -32700
    assert response["id"] is None

def test_method_not_found_error(server, capture_stdout):
    """PY-02: Unregistered method returns -32601."""
    request = json.dumps({"jsonrpc":"2.0","method":"nonexistent","params":{},"id":2})
    sys.stdin = io.StringIO(request)
    server.run()
    response = json.loads(capture_stdout.getvalue().strip())
    assert response["error"]["code"] == -32601
    assert "nonexistent" in response["error"]["message"]

def test_handler_exception_returns_error(server, capture_stdout):
    """PY-02: Handler exception returns -32000 with message."""
    server.register("crash", lambda p: 1/0)  # Will raise ZeroDivisionError
    request = json.dumps({"jsonrpc":"2.0","method":"crash","params":{},"id":3})
    sys.stdin = io.StringIO(request)
    server.run()
    response = json.loads(capture_stdout.getvalue().strip())
    assert response["error"]["code"] == -32000
    assert "division by zero" in response["error"]["message"]

def test_notification_produces_no_response(server, capture_stdout):
    """PY-02: Notification (no id) produces no output."""
    request = json.dumps({"jsonrpc":"2.0","method":"echo","params":{"msg":"silent"}})
    sys.stdin = io.StringIO(request)
    server.run()
    output = capture_stdout.getvalue().strip()
    assert output == "", f"Expected no output for notification, got: {output}"

def test_blank_line_skipped(server, capture_stdout):
    """PY-02: Blank lines are silently ignored."""
    input_data = "\n  \n" + json.dumps({"jsonrpc":"2.0","method":"echo","params":{"x":1},"id":4})
    sys.stdin = io.StringIO(input_data)
    server.run()
    response = json.loads(capture_stdout.getvalue().strip())
    assert response["result"] == {"x": 1}
```

**test_model.py** — CLIP model tests (note: use mocking to avoid downloading 605MB):
```python
import os, sys, pytest
from unittest.mock import patch, MagicMock
import torch

# Need sidecar on path
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..'))

def test_embedding_dimensions(temp_image, mock_model):
    """PY-01: generate_embedding returns 512-dim list."""
    with patch('model.load_model', return_value=mock_model):
        from model import generate_embedding
        embedding = generate_embedding(temp_image)
        assert isinstance(embedding, list)
        assert len(embedding) == 512, f"Expected 512, got {len(embedding)}"
        assert all(isinstance(v, float) for v in embedding)

def test_embedding_l2_normalized(temp_image, mock_model):
    """PY-01: Embedding is L2-normalized (norm ~1.0)."""
    # Use a mock that returns non-unit vectors so we verify normalization
    class NonUnitModel:
        def eval(self): return self
        def encode_image(self, t):
            return torch.tensor([[2.0] * 512])
    preprocess = mock_model[1]
    with patch('model.load_model', return_value=(NonUnitModel(), preprocess)):
        from model import generate_embedding
        embedding = generate_embedding(temp_image)
        import math
        norm = math.sqrt(sum(v*v for v in embedding))
        assert abs(norm - 1.0) < 1e-4, f"L2 norm should be ~1.0, got {norm}"

def test_model_cached_on_second_load():
    """PY-01: load_model returns same instance on second call."""
    with patch('model._model', None), patch('model._preprocess', None):
        with patch('open_clip.create_model_and_transforms') as mock_create:
            mock_model = MagicMock()
            mock_preprocess = MagicMock()
            mock_create.return_value = (mock_model, None, mock_preprocess)
            
            from model import load_model
            m1, p1 = load_model()
            m2, p2 = load_model()
            
            assert m1 is m2, "Model should be cached"
            assert p1 is p2, "Preprocess should be cached"
            mock_create.assert_called_once()  # Only called once

def test_missing_file_raises_error():
    """PY-01: Nonexistent path raises FileNotFoundError."""
    from model import generate_embedding
    with pytest.raises(FileNotFoundError, match="Image not found"):
        generate_embedding("/nonexistent/path/image.jpg")

def test_oversized_file_rejected(temp_large_image):
    """PY-01: Images >50MB raise ValueError."""
    # Skip if temp_large_image is not actually >50MB
    if os.path.getsize(temp_large_image) < 50 * 1024 * 1024:
        pytest.skip("temp_large_image fixture did not create a sufficiently large file")
    from model import generate_embedding
    with pytest.raises(ValueError, match="too large"):
        generate_embedding(temp_large_image)
```
