## Task 1: Create PlumFlower rating component

<action>
Create a new inline SVG component in ImageCard.tsx for the 梅花印 (plum flower) rating:

```tsx
function PlumFlower({ filled }: { filled: boolean }) {
  return (
    <svg width="14" height="14" viewBox="0 0 14 14" fill="none">
      {[0, 72, 144, 216, 288].map(angle => (
        <ellipse
          key={angle}
          cx="7" cy="3"
          rx="2.5" ry="3.5"
          transform={`rotate(${angle} 7 7)`}
          fill={filled ? 'var(--color-accent)' : 'none'}
          stroke={filled ? 'none' : 'rgba(139,115,75,0.15)'}
          strokeWidth="0.8"
        />
      ))}
      <circle
        cx="7" cy="7" r="1.5"
        fill={filled ? 'var(--color-accent)' : 'rgba(139,115,75,0.10)'}
      />
    </svg>
  );
}
```

Usage in card: `<PlumFlower filled={star <= image.rating} />`
</action>

<acceptance_criteria>
- Rating uses 5 plum flower SVGs (not stars, not dots)
- Filled: gold accent color with 5 petals + center
- Empty: faint outline only
- Each flower is 14x14px
</acceptance_criteria>
