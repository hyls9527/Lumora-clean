    **Step 3 — Create the store:**

    Use the exact pattern from app-store.ts:
    ```typescript
    export const useEmbeddingStore = create<EmbeddingState>((set, get) => ({
      statuses: generateMockEmbeddingStatuses(200),
      batch: null,
      isGenerating: false,
      isLoading: false,
      error: null,

      getStatus: (imageId) => {
        const s = get()
        return s.statuses.get(imageId) || { imageId, status: 'pending' }
      },

      embeddedCount: () => {
        let count = 0
        get().statuses.forEach((s) => { if (s.status === 'embedded') count++ })
        return count
      },

      pendingCount: () => {
        let count = 0
        get().statuses.forEach((s) => { if (s.status === 'pending') count++ })
        return count
      },

      errorCount: () => {
        let count = 0
        get().statuses.forEach((s) => { if (s.status === 'error') count++ })
        return count
      },

      loadStatuses: async () => {
        set({ isLoading: true, error: null })
        try {
          const allIds = Array.from({ length: 200 }, (_, i) => `img-${i}`)
          const results = await getEmbeddingStatus(allIds)
          if (results.length > 0) {
            const map = new Map<string, EmbeddingStatus>()
            results.forEach((s) => map.set(s.imageId, s))
            set({ statuses: map, isLoading: false })
          } else {
            set({ isLoading: false })
          }
        } catch (err) {
          console.error("loadStatuses failed:", err)
          set({ error: String(err), isLoading: false })
        }
      },

      generateEmbeddings: async (ids) => {
        set({ isGenerating: true, error: null })
        try {
          const { batchId } = await apiGenerateEmbeddings(ids)
          set({
            batch: { batchId, total: ids.length, current: 0, status: 'generating' },
          })
        } catch (err) {
          console.error("generateEmbeddings failed:", err)
          set({ error: String(err), isGenerating: false })
        }
      },

      cancelGeneration: async () => {
        const { batch } = get()
        if (batch && batch.status === 'generating') {
          try {
            await cancelEmbeddingGeneration(batch.batchId)
          } catch (err) {
            console.error("cancelGeneration failed:", err)
          }
        }
        set({ batch: null, isGenerating: false })
      },
    }))
    ```

    **Step 4 — Implement simulated progress for generateEmbeddings:**

    After setting the batch state in generateEmbeddings, use `setInterval` to simulate progress (matching the pattern from DropZone.tsx:95-115 and ExportDialog.tsx:52-66):
    - Start an interval at 200ms
    - Each tick: increment batch.current by a random amount (1-5), clamp to batch.total
    - When batch.current >= batch.total: set batch.status = 'complete', isGenerating = false, clear interval
    - Store the interval ID so cancelGeneration can clear it
    - Use a ref-like approach: store the interval ID as a variable in the module scope (above the store create call), e.g., `let progressInterval: ReturnType<typeof setInterval> | null = null`

    The interval logic should be inside the `generateEmbeddings` action, after `set({ batch: ... })`:
    ```
    progressInterval = setInterval(() => {
      const s = get()
      if (!s.batch || s.batch.status !== 'generating') {
        if (progressInterval) { clearInterval(progressInterval); progressInterval = null }
        return
      }
      const increment = Math.floor(Math.random() * 5) + 1
      const next = Math.min(s.batch.current + increment, s.batch.total)
      if (next >= s.batch.total) {
        if (progressInterval) { clearInterval(progressInterval); progressInterval = null }
        set({
          batch: { ...s.batch, current: s.batch.total, status: 'complete' },
          isGenerating: false,
        })
      } else {
        set({
          batch: { ...s.batch, current: next },
        })
      }
    }, 200)
    ```

    Also update cancelGeneration to clear the interval:
    ```
    if (progressInterval) { clearInterval(progressInterval); progressInterval = null }
    ```
