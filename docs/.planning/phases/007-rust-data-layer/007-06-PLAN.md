```rust
use serde::{Deserialize, Serialize};
use serde_json::Value;
use tauri::{AppHandle, Manager};
use tauri_plugin_store::StoreExt;

/// All known setting keys — used for validation and reset.
const KNOWN_SETTINGS: &[&str] = &[
    "language",
    "theme",
    "gridColumns",
    "maxFileSizeMB",
    // Reserved for future phases per D-19:
    "thumbnailQuality",
    "maxConcurrency",
    "aiTimeout",
];

/// Default values for all settings. Used by reset_settings per D-20.
fn default_settings() -> Vec<(&'static str, serde_json::Value)> {
    vec![
        ("language", serde_json::json!("zh")),
        ("theme", serde_json::json!("light")),
        ("gridColumns", serde_json::json!(4)),
        ("maxFileSizeMB", serde_json::json!(200)),  // Per Claude's discretion in CONTEXT.md
        // Future defaults per D-19:
        ("thumbnailQuality", serde_json::json!(80)),
        ("maxConcurrency", serde_json::json!(4)),
        ("aiTimeout", serde_json::json!(30000)),
    ]
}

/// Validation rules per D-21. Returns Some(error_message) if invalid.
fn validate_setting(key: &str, value: &str) -> Result<(), String> {
    match key {
        "language" => {
            let v = value.to_lowercase();
            if v != "en" && v != "zh" {
                return Err(format!("language must be 'en' or 'zh', got '{}'", value));
            }
        }
        "theme" => {
            let v = value.to_lowercase();
            if v != "light" && v != "dark" {
                return Err(format!("theme must be 'light' or 'dark', got '{}'", value));
            }
        }
        "gridColumns" => {
            let n: u32 = value.parse().map_err(|_| format!("gridColumns must be a number, got '{}'", value))?;
            if n < 1 || n > 10 {
                return Err(format!("gridColumns must be 1-10, got {}", n));
            }
        }
        "maxFileSizeMB" => {
            let n: u32 = value.parse().map_err(|_| format!("maxFileSizeMB must be a number, got '{}'", value))?;
            if n < 1 || n > 500 {
                return Err(format!("maxFileSizeMB must be 1-500, got {}", n));
            }
        }
        "thumbnailQuality" => {
            let n: u32 = value.parse().map_err(|_| format!("thumbnailQuality must be a number, got '{}'", value))?;
            if n < 1 || n > 100 {
                return Err(format!("thumbnailQuality must be 1-100, got {}", n));
            }
        }
        "maxConcurrency" => {
            let n: u32 = value.parse().map_err(|_| format!("maxConcurrency must be a number, got '{}'", value))?;
            if n < 1 || n > 16 {
                return Err(format!("maxConcurrency must be 1-16, got {}", n));
            }
        }
        "aiTimeout" => {
            let n: u32 = value.parse().map_err(|_| format!("aiTimeout must be a number, got '{}'", value))?;
            if n < 5000 || n > 120000 {
                return Err(format!("aiTimeout must be 5000-120000 (ms), got {}", n));
            }
        }
        _ => {
            // Unknown key — allow but warn (extensibility)
            eprintln!("[settings] Warning: setting unknown key '{}'", key);
        }
    }
    Ok(())
}
```

**Key design notes:**
- Settings are stored as strings in the plugin-store and parsed for validation.
- `gridColumns` defaults to 4 (matching the frontend `useSettingsStore` default in `src/stores/settings-store.ts:17`).
- `maxFileSizeMB` defaults to 200MB per Claude's discretion.
- Future settings (`thumbnailQuality`, `maxConcurrency`, `aiTimeout`) have defaults and validation per D-19 but won't be consumed by UI until their respective phases.
- Unknown keys are allowed (with a warning) for forward-compatibility — a newer version of the app might write a setting that an older version doesn't know about. This prevents data loss on downgrade.

**`get_setting` command:**
```rust
#[tauri::command]
pub fn get_setting(
    app: AppHandle,
    key: String,
) -> Result<String, String> {
    let store = app
        .store("settings.json")
        .map_err(|e| format!("Failed to open store: {}", e))?;

    match store.get(&key) {
        Some(value) => {
            // Convert the JSON value to a plain string for the frontend
            match value.as_str() {
                Some(s) => Ok(s.to_string()),
                None => {
                    // For numeric values (gridColumns, maxFileSizeMB), serialize to string
                    Ok(value.to_string())
                }
            }
        }
        None => {
            // Return default if not set
            let defaults = default_settings();
            for (k, v) in &defaults {
                if k == &key {
                    match v.as_str() {
                        Some(s) => return Ok(s.to_string()),
                        None => return Ok(v.to_string()),
                    }
                }
            }
            Err(format!("No default for key: {}", key))
        }
    }
}
```

**`set_setting` command:**

```rust
#[tauri::command]
pub fn set_setting(
    app: AppHandle,
    key: String,
    value: String,
) -> Result<(), String> {
    // Validate per D-21
    validate_setting(&key, &value)?;

    let store = app
        .store("settings.json")
        .map_err(|e| format!("Failed to open store: {}", e))?;

    // Store as JSON string value
    store.set(key, serde_json::Value::String(value));

    // Auto-save is handled by the plugin's debounce mechanism
    // (RESEARCH.md: "auto-save with debounce")
    Ok(())
}
```

**`reset_settings` command (per D-20):**
```rust
#[tauri::command]
pub fn reset_settings(
    app: AppHandle,
) -> Result<(), String> {
    let store = app
        .store("settings.json")
        .map_err(|e| format!("Failed to open store: {}", e))?;

    for (key, default_value) in default_settings() {
        store.set(key.to_string(), default_value);
    }

    // Force immediate save (not debounced) for reset action
    store.save()
        .map_err(|e| format!("Failed to save store: {}", e))?;

    Ok(())
}
```

**Important note on tauri-plugin-store API:**
The `tauri-plugin-store` v2 API uses `app.store("filename.json")` which returns a `Store` that is managed by the plugin. The `set()` method writes to the store. The store auto-saves with debounce. `store.save()` forces an immediate write.

If the exact API differs (e.g., requires `app.get_store()` or uses `StoreBuilder`), consult the installed version's docs. The `cargo check` step will catch API mismatches.

Implement the one-time migration from browser localStorage to tauri-plugin-store. Per D-18: "首次启动自动迁移浏览器 localStorage 中的旧设置（语言、主题、网格列数）到新存储。用户无感升级。" (First launch auto-migrates old settings from localStorage to the new store. User-transparent upgrade.)

**Migration strategy (from RESEARCH.md Pattern 5):**
1. The frontend (Phase 9) reads old localStorage values and passes them as a JSON string to `migrate_legacy_settings`.
2. Rust parses the JSON and writes each known key to the plugin-store.
3. After migration, a `_migrated` flag is set in the store so migration never runs again.
4. Only keys that exist in `KNOWN_SETTINGS` are migrated — unknown keys from localStorage are ignored.

**`migrate_legacy_settings` command:**
```rust
/// Accepts a JSON string of old localStorage key-value pairs.
/// Only migrates keys listed in KNOWN_SETTINGS.
/// Idempotent: if _migrated flag is true, does nothing.
#[tauri::command]
pub fn migrate_legacy_settings(
    app: AppHandle,
    old_settings: String,
) -> Result<(), String> {
    let store = app
        .store("settings.json")
        .map_err(|e| format!("Failed to open store: {}", e))?;

    // Check if migration already ran
    if store.get("_migrated").is_some() {
        return Ok(());
    }

    // Parse the JSON string from the frontend
    // Expected format: { "language": "zh", "theme": "light", "gridColumns": "4" }
    let old: serde_json::Value = serde_json::from_str(&old_settings)
        .map_err(|e| format!("Invalid JSON from frontend: {}", e))?;

    if let Some(obj) = old.as_object() {
        for (key, value) in obj {
            // Only migrate known settings (ignore arbitrary localStorage keys)
            if KNOWN_SETTINGS.contains(&key.as_str()) {
                // Convert value to string for the store
                let value_str = match value {
                    serde_json::Value::String(s) => s.clone(),
                    other => other.to_string(),
                };

                // Validate the value before writing per D-21
                if let Err(e) = validate_setting(key, &value_str) {
                    eprintln!("[migrate] Skipping invalid value for '{}': {}", key, e);
                    continue;
                }

                store.set(key.clone(), serde_json::Value::String(value_str));
            }
        }
    }

    // Mark migration as complete (idempotent guard)
    store.set(
        "_migrated".to_string(),
        serde_json::Value::String("true".to_string()),
    );

    // Force save — migration must persist before the app continues
    store.save()
        .map_err(|e| format!("Failed to save migrated settings: {}", e))?;

    Ok(())
}
```

**What gets migrated:**
From old `localStorage` keys (per `src/lib/api/settings.ts`):
- `language` → store key `language`
- `theme` → store key `theme`
- `gridColumns` → store key `gridColumns`

These are the only three settings stored in localStorage by the current `useSettingsStore`. The `maxFileSizeMB` setting is NEW in Phase 7 and has no old value to migrate.

**The `_migrated` sentinel:**
This is a special store key that acts as a migration flag. It is NOT in `KNOWN_SETTINGS` and is NOT exposed to the frontend via `get_setting`. The `reset_settings` command does NOT clear it (resetting settings should not re-trigger migration).
