**Part A: `filesystem/paths.rs` — Data directory resolution (D-22, Claude's discretion)**

Implement portable data directory logic. Per D-22: data next to install directory. Per Claude's discretion: fallback chain for write permission.

```rust
use std::path::{Path, PathBuf};

/// Resolve the data directory with fallback.
/// 1. Try {executable_dir}/data/  (portable, per D-22)
/// 2. Fallback: {app_local_data_dir}/Lumora/data/  (via dirs crate)
/// Returns (data_dir, images_dir, thumbnails_dir)
pub fn resolve_data_dirs() -> Result<(PathBuf, PathBuf, PathBuf), std::io::Error> {
    // Try portable location first (next to executable)
    let exe_dir = std::env::current_exe()
        .and_then(|p| p.parent().map(|d| d.to_path_buf()))
        .unwrap_or_else(|_| PathBuf::from("."));

    let portable_data = exe_dir.join("data");

    // Check if we can write to the portable location
    if is_writable(&exe_dir) {
        return create_dirs(&portable_data);
    }

    // Fallback: Windows %LOCALAPPDATA%/Lumora/data
    if let Some(local_data) = dirs::data_local_dir() {
        let fallback_data = local_data.join("Lumora").join("data");
        eprintln!("[paths] Portable directory not writable, using fallback: {:?}", fallback_data);
        return create_dirs(&fallback_data);
    }

    // Last resort: ./data in working directory
    create_dirs(&PathBuf::from("data"))
}

fn is_writable(dir: &Path) -> bool {
    // Try creating a temp file to test write permission
    let test_file = dir.join(".lumora_write_test");
    match std::fs::write(&test_file, b"test") {
        Ok(()) => {
            let _ = std::fs::remove_file(&test_file);
            true
        }
        Err(_) => false,
    }
}

fn create_dirs(base: &Path) -> Result<(PathBuf, PathBuf, PathBuf), std::io::Error> {
    let images = base.join("images");
    let thumbnails = base.join("thumbnails");
    std::fs::create_dir_all(&images)?;
    std::fs::create_dir_all(&thumbnails)?;
    Ok((base.to_path_buf(), images, thumbnails))
}
```

**Part B: `filesystem/scanner.rs` — Recursive folder scan (D-14, D-15)**

```rust
use std::path::{Path, PathBuf};
use walkdir::WalkDir;

const IMAGE_EXTENSIONS: &[&str] = &[
    "jpg", "jpeg", "png", "webp", "gif", "svg", "bmp", "tiff", "tif",
];

pub struct ScanResult {
    pub image_paths: Vec<PathBuf>,
    pub skipped_count: u32,  // Non-image files and symlinks skipped
}

/// Recursively scan a directory for supported image files.
/// Skips symlinks per D-15. Skips non-image files per D-15.
pub fn scan_image_files(root: &Path) -> ScanResult {
    let mut image_paths = Vec::new();
    let mut skipped = 0u32;

    for entry in WalkDir::new(root)
        .follow_links(false)  // D-15: do not follow symlinks
        .into_iter()
        .filter_map(|e| e.ok())
    {
        if !entry.file_type().is_file() {
            continue;
        }

        let path = entry.path();
        let is_image = path
            .extension()
            .and_then(|ext| ext.to_str())
            .map(|ext| IMAGE_EXTENSIONS.contains(&ext.to_lowercase().as_str()))
            .unwrap_or(false);

        if is_image {
            image_paths.push(path.to_path_buf());
        } else {
            skipped += 1;
        }
    }

    ScanResult {
        image_paths,
        skipped_count: skipped,
    }
}
```

**Part C: `filesystem/import.rs` — Core import pipeline**

Implements the per-file import pipeline with all D-01 through D-15 behaviors:

```rust
use std::fs;
use std::path::{Path, PathBuf};
use crate::imaging;

pub struct ImportFileResult {
    pub success: bool,
    pub file_path: PathBuf,       // Final path in data/images/
    pub thumbnail_path: Option<PathBuf>,  // Path in data/thumbnails/ or None if failed
    pub sha256_hash: String,
    pub original_name: String,
    pub error: Option<String>,
}

/// Process a single file for import. This is the core pipeline.
pub fn import_single_file(
    source_path: &Path,
    images_dir: &Path,
    thumbnails_dir: &Path,
    max_file_size_bytes: u64,
) -> ImportFileResult {
    let original_name = source_path
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("unknown")
        .to_string();

    // 1. Check file size per D-09
    let file_size = match fs::metadata(source_path) {
        Ok(m) => m.len(),
        Err(e) => {
            return ImportFileResult {
                success: false,
                file_path: source_path.to_path_buf(),
                thumbnail_path: None,
                sha256_hash: String::new(),
                original_name,
                error: Some(format!("Cannot read file: {}", e)),
            };
        }
    };

    if file_size > max_file_size_bytes {
        return ImportFileResult {
            success: false,
            file_path: source_path.to_path_buf(),
            thumbnail_path: None,
            sha256_hash: String::new(),
            original_name,
            error: Some(format!("File too large: {} bytes (max: {})", file_size, max_file_size_bytes)),
        };
    }

    // 2. Determine if format conversion is needed (D-06: BMP/TIFF → PNG)
    let needs_conversion = needs_format_conversion(source_path);
    let dest_name = if needs_conversion {
        let stem = source_path.file_stem().and_then(|s| s.to_str()).unwrap_or("converted");
        format!("{}.png", stem)
    } else {
        original_name.clone()
    };

    // 3. Resolve name conflicts per D-12 (auto-suffix)
    let dest_path = resolve_name_conflict(&images_dir.join(&dest_name));

    // 4. Copy or convert file to data/images/ per D-08
    if needs_conversion {
        if let Err(e) = imaging::formats::convert_to_png(source_path, &dest_path) {
            return ImportFileResult {
                success: false,
                file_path: dest_path,
                thumbnail_path: None,
                sha256_hash: String::new(),
                original_name,
                error: Some(format!("BMP/TIFF conversion failed: {}", e)),
            };
        }
    } else {
        if let Err(e) = fs::copy(source_path, &dest_path) {
            return ImportFileResult {
                success: false,
                file_path: dest_path,
                thumbnail_path: None,
                sha256_hash: String::new(),
                original_name,
                error: Some(format!("File copy failed: {}", e)),
            };
        }
    }

    // 5. Compute SHA-256 hash per D-11
    let hash = match imaging::hash::compute_file_hash(&dest_path) {
        Ok(h) => h,
        Err(e) => {
            return ImportFileResult {
                success: false,
                file_path: dest_path,
                thumbnail_path: None,
                sha256_hash: String::new(),
                original_name,
                error: Some(format!("Hash computation failed: {}", e)),
            };
        }
    };

    // 6. Generate thumbnail per D-01, D-02, D-04
    // SVG files cannot be thumbnailed by the image crate — skip per RESEARCH.md Open Question #1
    let thumbnail_path = if is_svg(&dest_path) {
        None // SVG: no thumbnail, but import succeeds
    } else {
        let thumb_name = format!("{}.webp", dest_path.file_stem().and_then(|s| s.to_str()).unwrap_or("thumb"));
        let thumb_dest = resolve_name_conflict(&thumbnails_dir.join(&thumb_name));
        match imaging::thumbnail::generate_thumbnail(&dest_path, &thumb_dest, 512) {
            Ok(()) => Some(thumb_dest),
            Err(e) => {
                // D-03: Thumbnail failure does not block import
                eprintln!("[import] Thumbnail generation failed for {:?}: {}", dest_path, e);
                None
            }
        }
    };

    ImportFileResult {
        success: true,
        file_path: dest_path,
        thumbnail_path,
        sha256_hash: hash,
        original_name,
        error: None,
    }
}

/// Check if a file needs BMP/TIFF→PNG conversion per D-06
fn needs_format_conversion(path: &Path) -> bool {
    path.extension()
        .and_then(|e| e.to_str())
        .map(|e| matches!(e.to_lowercase().as_str(), "bmp" | "tiff" | "tif"))
        .unwrap_or(false)
}

/// Check if a file is SVG (cannot be thumbnailed)
fn is_svg(path: &Path) -> bool {
    path.extension()
        .and_then(|e| e.to_str())
        .map(|e| e.to_lowercase().as_str() == "svg")
        .unwrap_or(false)
}

/// Resolve file name conflicts per D-12.
/// `IMG_0001.jpg` → `IMG_0001_2.jpg` → `IMG_0001_3.jpg` ...
pub fn resolve_name_conflict(path: &Path) -> PathBuf {
    if !path.exists() {
        return path.to_path_buf();
    }

    let stem = path.file_stem().and_then(|s| s.to_str()).unwrap_or("file");
    let ext = path.extension().and_then(|s| s.to_str()).unwrap_or("");
    let parent = path.parent().unwrap_or(Path::new("."));

    let mut counter = 2u32;
    loop {
        let new_name = if ext.is_empty() {
            format!("{}_{}", stem, counter)
        } else {
            format!("{}_{}.{}", stem, counter, ext)
        };
        let new_path = parent.join(&new_name);
        if !new_path.exists() {
            return new_path;
        }
        counter += 1;
    }
}
```

**Part A: `imaging/thumbnail.rs` — 512px WebP thumbnails (D-01, D-02, D-04)**

Implement using `image` crate with Lanczos3 filter per RESEARCH.md:

```rust
use image::imageops::FilterType;
use std::fs::File;
use std::io::BufWriter;
use std::path::Path;

/// Generate a thumbnail at max_dimension (512px) as WebP.
/// Maintains aspect ratio. Uses Lanczos3 for quality.
/// Per D-02: targets ~20KB at 512px WebP.
pub fn generate_thumbnail(
    input_path: &Path,
    output_path: &Path,
    max_dimension: u32,
) -> Result<(), Box<dyn std::error::Error>> {
    let img = image::open(input_path)?;
    let (w, h) = img.dimensions();

    // Maintain aspect ratio, fit within max_dimension x max_dimension
    let scale = max_dimension as f64 / w.max(h) as f64;
    let new_w = (w as f64 * scale).round() as u32;
    let new_h = (h as f64 * scale).round() as u32;

    // Use Lanczos3 for highest-quality downscaling
    let thumb = img.resize_exact(new_w.max(1), new_h.max(1), FilterType::Lanczos3);

    // Write as WebP
    let file = File::create(output_path)?;
    let writer = BufWriter::new(file);
    thumb.write_to(&mut std::io::BufWriter::new(writer), image::ImageFormat::WebP)?;

    Ok(())
}
```

**Part C: `imaging/hash.rs` — SHA-256 dedup (D-11)**

```rust
use sha2::{Sha256, Digest};
use std::fs;
use std::io::Read;
use std::path::Path;

/// Compute SHA-256 hash of a file's contents.
/// Used for duplicate detection per D-11.
/// Streams the file to avoid loading large images into memory.
pub fn compute_file_hash(path: &Path) -> Result<String, std::io::Error> {
    let mut file = fs::File::open(path)?;
    let mut hasher = Sha256::new();
    let mut buffer = [0u8; 8192];

    loop {
        let bytes_read = file.read(&mut buffer)?;
        if bytes_read == 0 {
            break;
        }
        hasher.update(&buffer[..bytes_read]);
    }

    let hash = hasher.finalize();
    Ok(format!("{:x}", hash))
}

#[cfg(test)]
mod tests {
    use super::*;
    use std::io::Write;

    #[test]
    fn test_file_hash_deterministic() {
        let dir = tempfile::tempdir().unwrap();
        let file_path = dir.path().join("test.txt");
        let mut f = fs::File::create(&file_path).unwrap();
        f.write_all(b"hello world").unwrap();

        let h1 = compute_file_hash(&file_path).unwrap();
        let h2 = compute_file_hash(&file_path).unwrap();
        assert_eq!(h1, h2);
        assert_eq!(h1.len(), 64); // SHA-256 produces 64 hex chars
    }

    #[test]
    fn test_different_files_different_hash() {
        let dir = tempfile::tempdir().unwrap();
        let f1 = dir.path().join("a.txt");
        let f2 = dir.path().join("b.txt");
        fs::write(&f1, b"hello").unwrap();
        fs::write(&f2, b"world").unwrap();

        let h1 = compute_file_hash(&f1).unwrap();
        let h2 = compute_file_hash(&f2).unwrap();
        assert_ne!(h1, h2);
    }
}
```

```rust
// Read max file size from tauri-plugin-store at the start of import (NOT from AppConfig)
// Per D-09: user-configurable via Settings UI, stored in plugin-store.
// Reading at import time ensures the latest user setting is always respected.
let max_file_size_bytes: u64 = {
    use tauri_plugin_store::StoreExt;
    app_handle
        .store("settings.json")
        .ok()
        .and_then(|store| store.get("maxFileSizeMB"))
        .and_then(|v| v.as_str().map(|s| s.parse::<u64>().ok()))
        .flatten()
        .unwrap_or(200)  // Default 200MB per Claude's discretion
        * 1024 * 1024
};

// Run the full import pipeline (file copy, convert, hash, thumbnail)
let import_result = filesystem::import::import_single_file(
    path,
    &app_state.images_dir,
    &app_state.thumbnails_dir,
    max_file_size_bytes,
);

if !import_result.success {
    if let Some(err) = &import_result.error {
        if err.contains("File too large") {
            result.skipped_too_large += 1;
        } else {
            result.errors.push(err.clone());
        }
    }
    let _ = app_for_events.emit("import-progress", ImportProgress {
        current: i as u32 + 1,
        total,
        file_name: path_str.clone(),
    });
    continue;
}

// Check for duplicates per D-11
if db::images::is_duplicate(conn, &import_result.sha256_hash)? {
    result.skipped_duplicates += 1;
    // Clean up copied file since it's a duplicate
    let _ = std::fs::remove_file(&import_result.file_path);
    if let Some(ref tp) = import_result.thumbnail_path {
        let _ = std::fs::remove_file(tp);
    }
    let _ = app_for_events.emit("import-progress", ImportProgress {
        current: i as u32 + 1,
        total,
        file_name: path_str.clone(),
    });
    continue;
}

// Read image dimensions from the copied/converted file
let (width, height) = match image::image_dimensions(&import_result.file_path) {
    Ok(dims) => dims,
    Err(e) => {
        result.errors.push(format!("Cannot read dimensions for {}: {}", path_str, e));
        let _ = app_for_events.emit("import-progress", ImportProgress { ... });
        continue;
    }
};

let file_size = import_result.file_path
    .metadata()
    .map(|m| m.len() as i64)
    .unwrap_or(0);

// Determine format from the final file (BMP/TIFF become PNG)
let format = import_result.file_path
    .extension()
    .and_then(|e| e.to_str())
    .map(|e| e.to_lowercase())
    .unwrap_or_else(|| "unknown".to_string());

let thumbnail_path_str = import_result.thumbnail_path
    .as_ref()
    .and_then(|p| p.to_str())
    .map(String::from);

let id = uuid::Uuid::new_v4().to_string();

let new_image = db::models::NewImage {
    id: id.clone(),
    file_path: import_result.file_path.to_string_lossy().to_string(),
    original_name: import_result.original_name,
    width: width as i32,
    height: height as i32,
    file_size,
    format,
    thumbnail_path: thumbnail_path_str,
    sha256_hash: Some(import_result.sha256_hash),
};
```
