**Part A: Create `src/components/ImportProgressBar.tsx`**

New component per 007-UI-SPEC.md element #1. Reuses the visual treatment from `BatchEmbeddingBar.tsx` (Phase 004).

```tsx
import { useTranslation } from "@/lib/i18n"
import { cn } from "@/lib/utils"

interface ImportProgressBarProps {
  current: number
  total: number
  onCancel?: () => void
  className?: string
}

export function ImportProgressBar({ current, total, onCancel, className }: ImportProgressBarProps) {
  const { t } = useTranslation()
  const percent = total > 0 ? Math.round((current / total) * 100) : 0

  if (total === 0) return null

  return (
    <div className={cn(
      "flex items-center gap-3 px-4 py-2 bg-surface border-b border-border-subtle",
      className
    )}>
      {/* Progress bar track */}
      <div className="flex-1 rounded-[2px] bg-accent/20 overflow-hidden h-1.5">
        <div
          className="h-full rounded-[2px] bg-accent transition-all duration-200 ease-out"
          style={{ width: `${percent}%` }}
        />
      </div>

      {/* Progress text */}
      <span className="font-sans text-[12px] text-text-muted tabular-nums whitespace-nowrap">
        {t("import.progressLabel")
          .replace("{current}", String(current))
          .replace("{total}", String(total))}
      </span>

      {/* Cancel button */}
      {onCancel && (
        <button
          onClick={onCancel}
          className="text-[12px] text-text-muted hover:text-text-secondary
                     transition-all duration-200 ease-out font-sans whitespace-nowrap"
        >
          {t("import.cancel")}
        </button>
      )}
    </div>
  )
}
```
