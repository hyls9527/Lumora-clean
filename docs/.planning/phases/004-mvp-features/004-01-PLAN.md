## Task 2: Create VirtualizedGrid component

<action>
Create a new component `src/components/VirtualizedGrid.tsx` that uses react-window to render a virtualized grid:

```tsx
import { FixedSizeGrid as Grid } from 'react-window';
import { ImageCard } from './ImageCard';
import { type Image } from '@/lib/mock-data';

interface VirtualizedGridProps {
  images: Image[];
  columns: number;
  columnWidth: number;
  rowHeight: number;
  height: number;
  width: number;
}

export function VirtualizedGrid({ 
  images, 
  columns, 
  columnWidth, 
  rowHeight, 
  height, 
  width 
}: VirtualizedGridProps) {
  const rowCount = Math.ceil(images.length / columns);
  
  const Cell = ({ columnIndex, rowIndex, style }: any) => {
    const index = rowIndex * columns + columnIndex;
    if (index >= images.length) return null;
    
    return (
      <div style={style}>
        <div className="p-1">
          <ImageCard image={images[index]} />
        </div>
      </div>
    );
  };
  
  return (
    <Grid
      columnCount={columns}
      columnWidth={columnWidth}
      height={height}
      rowCount={rowCount}
      rowHeight={rowHeight}
      width={width}
    >
      {Cell}
    </Grid>
  );
}
```
</action>

<acceptance_criteria>

- Only visible items are rendered in the DOM
- Scrolling is smooth
- TypeScript compiles without errors
