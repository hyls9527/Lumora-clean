# Plan 006-03: Error Boundaries + Polish

## Task 1: Create page-level ErrorBoundary wrapper

<action>
Create a reusable ErrorBoundary wrapper for pages:

```typescript
// src/components/PageErrorBoundary.tsx
import { ErrorBoundary } from "./ErrorBoundary"

export function PageErrorBoundary({ children }: { children: React.ReactNode }) {
  return (
    <ErrorBoundary
      fallback={
        <div className="flex items-center justify-center h-full">
          <div className="text-center">
            <p className="font-serif text-[16px] text-text-muted mb-2">
              此页尚有未竟之事
            </p>
            <button
              onClick={() => window.location.reload()}
              className="font-serif text-[13px] text-accent hover:underline"
            >
              重新加载
            </button>
          </div>
        </div>
      }
    >
      {children}
    </ErrorBoundary>
  )
}
```
</action>

<acceptance_criteria>
- Reusable page-level error boundary
- Poetic fallback text
- Reload button works
</acceptance_criteria>
