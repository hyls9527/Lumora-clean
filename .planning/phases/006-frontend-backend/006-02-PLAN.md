Update src-tauri/src/db.rs:

```rust
pub fn toggle_favorite(&self, image_id: i64) -> Result<bool> {
    let current: i32 = self.conn.query_row(
        "SELECT favorite FROM images WHERE id = ?1",
        [image_id],
        |row| row.get(0),
    )?;
    let new_val = if current == 0 { 1 } else { 0 };
    self.conn.execute(
        "UPDATE images SET favorite = ?1 WHERE id = ?2",
        [new_val, image_id],
    )?;
    Ok(new_val != 0)
}
```
</action>

<acceptance_criteria>

- Returns new favorite state

Update src-tauri/src/db.rs:

```rust
pub fn search_images(&self, query: &str) -> Result<Vec<ImageRecord>> {
    let mut stmt = self.conn.prepare(
        "SELECT id, file_path, file_hash, file_size_kb, width, height, format, 
                created_at, rating, favorite, metadata_json
         FROM images WHERE deleted = 0 AND (
           file_path LIKE '%' || ?1 || '%'
           OR metadata_json LIKE '%' || ?1 || '%'
         )
         ORDER BY created_at DESC LIMIT 100"
    )?;
    
    let images = stmt.query_map([query], |row| {
        Ok(ImageRecord {
            id: row.get(0)?,
            file_path: row.get(1)?,
            file_hash: row.get(2)?,
            file_size_kb: row.get(3)?,
            width: row.get(4)?,
            height: row.get(5)?,
            format: row.get(6)?,
            created_at: row.get(7)?,
            rating: row.get(8)?,
            favorite: row.get::<_, i32>(9)? != 0,
            metadata_json: row.get(10)?,
        })
    })?.collect::<Result<Vec<_>>>()?;
    
    Ok(images)
}
```
</action>

<acceptance_criteria>
- Search works on file_path and metadata
- Returns matching images
- Limited to 100 results
</acceptance_criteria>
